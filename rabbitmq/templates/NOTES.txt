To access for outside the cluster, perform the following steps:

{{- if contains "NodePort" .Values.service.type }}

    export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")

Obtain the NodePort IP and ports:

    {{- if .Values.rabbitmq.tls.enabled }}

    export NODE_PORT_AMQPS=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[2].nodePort}" services {{ include "rabbitmq.fullname" . }})
    export NODE_PORT_MANAGEMENT_SSL=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[3].nodePort}" services {{ include "rabbitmq.fullname" . }})

To Access the RabbitMQ AMQPS port:

    echo "URL : amqps://$NODE_IP:$NODE_PORT_AMQPS/"

To Access the RabbitMQ Management interface SSL:

    echo "URL : https://$NODE_IP:$NODE_PORT_MANAGEMENT_SSL/"

    {{- else }}

    export NODE_PORT_AMQP=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "rabbitmq.fullname" . }})
    export NODE_PORT_MANAGEMENT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[1].nodePort}" services {{ include "rabbitmq.fullname" . }})

To Access the RabbitMQ AMQP port:

    echo "URL : amqp://$NODE_IP:$NODE_PORT_AMQP/"

To Access the RabbitMQ Management interface:

    echo "URL : http://$NODE_IP:$NODE_PORT_MANAGEMENT/"
